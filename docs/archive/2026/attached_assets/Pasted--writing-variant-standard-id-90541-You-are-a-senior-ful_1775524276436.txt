export function restoreOfflineSession() {
  try {
    const raw = localStorage.getItem("vt_session");
    if (!raw) return null;

    const data = JSON.parse(raw);

    // Validate structure
    if (!data || !data.token || !data.tokenExp) return null;

    // HARD FAIL — expired token blocks access
    if (Date.now() > data.tokenExp) {
      return null;
    }

    // Secondary TTL (24h safety)
    const MAX_OFFLINE_AGE = 24 * 60 * 60 * 1000;
    if (!data.lastActiveAt || Date.now() - data.lastActiveAt > MAX_OFFLINE_AGE) {
      return null;
    }

    return data;
  } catch {
    return null;
  }
}
