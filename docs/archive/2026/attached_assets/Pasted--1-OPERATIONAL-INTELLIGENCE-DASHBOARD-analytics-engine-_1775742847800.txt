type TrendPoint = { date: string; count: number };

export function computeUsageTrends(logs: ScanLog[]): TrendPoint[] {
  const map = new Map<string, number>();

  for (const log of logs) {
    const date = log.timestamp.split("T")[0];
    map.set(date, (map.get(date) ?? 0) + 1);
  }

  return Array.from(map.entries())
    .map(([date, count]) => ({ date, count }))
    .sort((a, b) => a.date.localeCompare(b.date));
}

export function detectOutliers(equipment: Equipment[]) {
  const now = Date.now();

  return equipment.filter((eq) => {
    const lastSeen = eq.lastSeen ? Date.parse(eq.lastSeen) : 0;
    return now - lastSeen > 72 * 60 * 60 * 1000;
  });
}

🧠 2. BACKEND DESIGN (IDEMPOTENT + VERSIONED)

example endpoint

app.post("/api/equipment/:id/scan", async (req, res) => {
  const { id } = req.params;
  const { status } = req.body;
  const clientTimestamp = req.headers["x-client-timestamp"];

  const existing = await db.equipment.find(id);

  if (existing.lastClientTimestamp >= clientTimestamp) {
    return res.status(409).json({ error: "conflict", server: existing });
  }

  const updated = await db.equipment.update(id, {
    status,
    lastClientTimestamp: clientTimestamp,
    version: existing.version + 1,
  });

  res.json(updated);
});

state reconstruction

export function rebuildState(events: Event[]): Record<string, Equipment> {
  const state: Record<string, Equipment> = {};

  for (const event of events) {
    switch (event.type) {
      case "equipment_created":
        state[event.payload.id] = event.payload;
        break;

      case "equipment_scanned":
        state[event.payload.id].status = event.payload.status;
        break;

      case "equipment_checked_out":
        state[event.payload.id].checkedOutById = event.payload.userId;
        break;
    }
  }

  return state;
}

export function fuzzySearch(items: Equipment[], query: string) {
  const q = query.toLowerCase();

  return items.filter((item) =>
    item.name.toLowerCase().includes(q) ||
    (item.location || "").toLowerCase().includes(q)
  );
}
