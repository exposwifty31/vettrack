async function handleOptimisticMutation<T>({
  endpoint,
  method,
  body,
  optimisticUpdate,
  cacheUpdate,
  syncType,
}: {
  endpoint: string;
  method: string;
  body?: unknown;
  optimisticUpdate: () => Promise<T>;
  cacheUpdate?: (result: T) => void;
  syncType: PendingSyncType;
}): Promise<T> {
  const clientTimestamp = Date.now();

  try {
    const result = await request<T>(endpoint, {
      method,
      // Do NOT fall back to "{}" — preserve absence of body when none was intended
      body: body !== undefined ? JSON.stringify(body) : undefined,
      headers: { "X-Client-Timestamp": String(clientTimestamp) },
    });

    cacheUpdate?.(result);
    return result;
  } catch (err) {
    if (isNetworkError(err)) {
      const optimistic = await optimisticUpdate();

      await addPendingSync({
        type: syncType,
        endpoint,
        method,
        // Store empty string (not "{}") when body is absent
        body: body !== undefined ? JSON.stringify(body) : "",
        createdAt: new Date(),
        retries: 0,
        status: "pending",
        clientTimestamp,
        // Guard against optimisticUpdate returning undefined/null
        optimisticData: optimistic ? JSON.stringify(optimistic) : undefined,
      });

      return optimistic;
    }
    throw err;
  }
}

import type { Equipment, CriticalItem, UserEquipmentGroup } from "@/types";

export function computeDashboardData(equipment: Equipment[]) {
  const now = Date.now();

  let available = 0;
  let inUse = 0;
  let issues = 0;
  let missing = 0;

  let missingCost = 0;
  let issueCost = 0;

  const criticalItems: CriticalItem[] = [];
  const userMap = new Map<string, UserEquipmentGroup>();
  const locationMap = new Map<string, number>();

  for (const eq of equipment) {
    const lastSeenTs = eq.lastSeen ? Date.parse(eq.lastSeen) : 0;
    const isInUse = !!eq.checkedOutById;
    const isIssue = eq.status === "issue";
    const isAvailable = eq.status === "ok" && !isInUse;

    // FIX: checked-out equipment is with someone — never flag as missing
    const isMissing =
      !isInUse && (!lastSeenTs || now - lastSeenTs > 86400000);

    if (isAvailable) available++;
    if (isInUse) inUse++;
    if (isIssue) issues++;
    if (isMissing) missing++;

    if (isMissing) missingCost += 500;
    if (isIssue) issueCost += 75;

    if (isIssue || isMissing) {
      criticalItems.push({
        id: eq.id,
        name: eq.name,
        reason: isIssue
          ? "Active Issue"
          : eq.lastSeen
          ? "Not seen in 24+ hours"
          : "Never scanned",
        location: eq.location,
        status: isIssue ? "issue" : "missing",
      });
    }

    if (isInUse && eq.checkedOutById) {
      if (!userMap.has(eq.checkedOutById)) {
        userMap.set(eq.checkedOutById, {
          userId: eq.checkedOutById,
          userEmail: eq.checkedOutByEmail || eq.checkedOutById,
          items: [],
        });
      }
      userMap.get(eq.checkedOutById)!.items.push(eq);
    }

    const loc = eq.location || "Unknown";
    locationMap.set(loc, (locationMap.get(loc) ?? 0) + 1);
  }

  const userGroups = Array.from(userMap.values()).sort(
    (a, b) => b.items.length - a.items.length
  );

  const locationGroups = Array.from(locationMap.entries())
    .map(([location, count]) => ({ location, count }))
    .sort((a, b) => b.count - a.count);

  const total = equipment.length;
  const operational = available + inUse;

  return {
    counts: { available, inUse, issues, missing },
    criticalItems,
    userGroups,
    locationGroups,
    costEstimate: {
      missingCost,
      issueCost,
      total: missingCost + issueCost,
    },
    operationalPercent:
      total === 0 ? 100 : Math.round((operational / total) * 100),
  };
}
